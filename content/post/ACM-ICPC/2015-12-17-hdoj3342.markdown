    const int N=1E2+7;
    
    int n,m;
    
    bool v[N][N];
    int in[N];
    
    void topo()
    {
        queue<int>q;
    
        for ( int i = 0 ; i < n ; i++)
        {
    	if (in[i]==0) q.push(i);
        }
    
        int cnt = 0 ;
        while (!q.empty())
        {
    	cnt++;
    	int u = q.front(); q.pop();
    
    	for ( int i = 0 ; i< n ; i++)
    	{
    	    if (!v[u][i]) continue;
    
    	    in[i]--;
    	    if (in[i]==0)
    	    {
    		q.push(i);
    	    }
    	}
        }
        if (cnt==n)
        {
    	puts("YES");
        }
        else
        {
    	puts("NO");
        }
    }

    	    ms(v,false);

    	    while (m--)
    	    {
    		int x,y;
    		scanf("%d %d",&x,&y);
    		if (v[x][y]) continue ; //重边?

    		in[y]++;
    	    }
    
    	    topo();
