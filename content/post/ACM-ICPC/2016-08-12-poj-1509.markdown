    int minRep(char *s)
    {
        int n = strlen(s);
        int i = 0;
        int j = 1;
        int k = 0;
        while (i<n&&j<n&&k<n)
        {
    	int t = s[(i+k)%n]-s[(j+k)%n];
    	if (t==0) k++;
    	else 
    	{
    	    if (t>0)
    		i+=k+1;
    	    else j+=k+1;
    	    if (i==j) j++;
    	    k = 0 ;
    	}
        }
        return i<j?i:j;
    }
